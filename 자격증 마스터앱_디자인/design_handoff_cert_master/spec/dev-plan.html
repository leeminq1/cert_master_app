  q_id        INTEGER PRIMARY KEY AUTOINCREMENT,

CREATE INDEX idx_q_cert ON questions(cert_id);
CREATE INDEX idx_q_tags ON questions(tags);

LOGIC 1 · 자기 평가 → 복습 스케줄

SM-2 변형 복습 알고리즘

사용자가 풀이 후 다시 애매 확실 중 하나 선택. 이 등급(0/1/2)으로 다음 복습 시점을 자동 계산.

// scheduleReview(qState, grade) → updated qState
function scheduleReview(s, grade) {
  const now = Date.now();

  // 1. ease factor 조정 (1.3 ~ 2.8)
  let ease = s.ease + (0.1 - (2-grade) * 0.18);
  ease = Math.max(1.3, Math.min(2.8, ease));

  // 2. interval 계산 (단위: 일)
  let iv;
  if (grade === 0) {           // 다시 → 10분 후
    iv = 10/1440;
    s.wrong_count++;
  } else if (s.interval_days < 1) iv = 1;
  else if (s.interval_days < 3) iv = 3;
  else iv = Math.round(s.interval_days * ease);

  // 3. mastery_level (3회 연속 '확실' = 완료)
  if (grade === 2 && s.last_grade === 2) s.mastery_level = 2;
  else if (grade > 0) s.mastery_level = 1;
  else s.mastery_level = 0;

  return {
    ...s, ease, interval_days: iv,
    last_grade: grade,
    next_review_at: now + iv * 86400000
  };
}

등급별 동작

등급다음 노출오답카운트마스터리
0 다시10분 후+1→ 0
1 애매1·3·N일 후유지→ 1
2 확실N×ease일 후유지2회연속→2

복습 큐 쿼리

SELECT q.* FROM questions q
  JOIN q_state s ON s.q_id=q.q_id
 WHERE q.cert_id = ?
   AND s.next_review_at <= strftime('%s','now')*1000
   AND s.mastery_level < 2
 ORDER BY s.next_review_at ASC
 LIMIT ?;

// 매일 자정에 호출 또는 앱 진입 시 lazy 계산
function computeStreak() {
  let streak = 0;
  const today = ymd(new Date());
  for (let d = 0; d < 365; d++) {
    const day = ymd(addDays(new Date(), -d));
    const row = db.get("SELECT * FROM daily_activity WHERE date_ymd=?", day);
    if (row?.goal_met) streak++;
    else if (d === 0) continue;  // 오늘은 미달성이어도 어제까지로
    else break;
  }
  return streak;
}

점수 예측 (실제 시험 60문항 기준)

// 카테고리별 정답률 가중평균 → 100점 환산
function predictScore(certId) {
  const rows = db.all(`
    SELECT q.category,
           COUNT(*) c,
           SUM(CASE WHEN a.grade=2 THEN 1 ELSE 0 END) ok
    FROM attempts a
    JOIN questions q USING(q_id)
    WHERE q.cert_id=?
      AND a.attempted_at > ? /* 최근 30일 */
    GROUP BY q.category
  `, certId, Date.now() - 30*86400000);

  if (!rows.length) return null;
  const wAvg = rows.reduce((s,r) => s + (r.ok/r.c), 0) / rows.length;
  return Math.round(wAvg * 100);
}

합격권 ≥ 60 60점 이상 시 "합격권" 배지 표시. 일주일 단위 변화량 표기.

    timestamp: nextOccurrence(hour, minute).getTime(),
    repeatFrequency: RepeatFrequency.DAILY

    android: { channelId: 'reminder', smallIcon: 'ic_stat' },
    ios: { sound: 'default' }

}

유사 개념 묶음 (Related)

현재 문제의 태그 교집합으로 같은 자격증 내 유사 문제 정렬.

function findRelated(qId, k=5) {
  const q = db.get('SELECT * FROM questions WHERE q_id=?', qId);
  const tags = q.tags.split(',');
  return db.all(`
    SELECT q.*, (
      ${tags.map(()=>"(',' || q.tags || ',') LIKE ?").join(' + ')}
    ) AS overlap
    FROM questions q
    WHERE q.cert_id = ?
      AND q.q_id != ?
    ORDER BY overlap DESC, RANDOM()
    LIMIT ?
  `, ...tags.map(t=>`%,${t},%`), q.cert_id, qId, k);
}

상위 결과로 "유사 개념 5문항" 카드 구성. 1분 핵심 카드도 동일 로직으로 8개 묶음 자동 생성 가능.
